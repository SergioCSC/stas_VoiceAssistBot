Передаем FFMPEG_BINARY в качестве первого аргумента команды subprocess.run(), а затем передаем остальные аргументы команды ffmpeg в виде списка command.

# Запуск команды для конвертации аудио
command = ["ffmpeg", "-i", "input.mp3", "output.wav"]
subprocess.run(command, check=True)

# Запуск команды для извлечения аудио из видео
command = ["ffmpeg", "-i", "input.mp4", "-vn", "-acodec", "copy", "output.m4a"]
subprocess.run(command, check=True)

async def send_echo(message: Message, bot: Bot):
    try:
        file_id = message.voice.file_id if message.voice else message.audio.file_id
        await message.reply(text=f"file info \n {file_id}")
        file = await bot.get_file(file_id)
        file_path = file.file_path
        await bot.download_file(file_path, "some_file.mp3")
